**Goal:** 사이드바, 분할 패널, 탭, 키보드 단축키가 완전히 동작하는 UI. 사용자가 워크스페이스를 만들고, 터미널을 분할하고, 탭을 전환하고, 단축키로 조작할 수 있어야 한다.

**Tech Stack:** React 18.3, CSS Grid/Flexbox, @dnd-kit (드래그), Electron before-input-event (단축키)

### preload/index.ts 수정

```typescript
// surfaceId → ptyId 매핑 (P2-BUG-7)
const surfacePtyMap = new Map<string, string>();

contextBridge.exposeInMainWorld('ptyBridge', {
  spawn(surfaceId: string, options?: { shell?: string; cwd?: string; cols?: number; rows?: number }) {
    const result = bridge.spawn(options);
    surfacePtyMap.set(surfaceId, result.id);    // surfaceId → ptyId 매핑 저장
    ipcRenderer.send(IPC_CHANNELS.PTY_METADATA, { surfaceId, ptyId: result.id, pid: result.pid });
    return result;
  },
  write(surfaceId: string, data: string) {
    const ptyId = surfacePtyMap.get(surfaceId);
    if (ptyId) bridge.write(ptyId, data);
  },
  resize(surfaceId: string, cols: number, rows: number) {
    const ptyId = surfacePtyMap.get(surfaceId);
    if (ptyId) bridge.resize(ptyId, cols, rows);
  },
  kill(surfaceId: string) {
    const ptyId = surfacePtyMap.get(surfaceId);
    if (ptyId) { bridge.kill(ptyId); surfacePtyMap.delete(surfaceId); }
  },
  has(surfaceId: string) {
    const ptyId = surfacePtyMap.get(surfaceId);
    return ptyId ? bridge.has(ptyId) : false;
  },
  onData(surfaceId: string, callback: (data: string) => void) {
    const ptyId = surfacePtyMap.get(surfaceId);
    if (ptyId) return bridge.onData(ptyId, callback);
  },
  onExit(surfaceId: string, callback: (e: { exitCode: number; signal?: number }) => void) {
    const ptyId = surfacePtyMap.get(surfaceId);
    if (ptyId) return bridge.onExit(ptyId, callback);
  },
  getAvailableShells() { return bridge.getAvailableShells(); },
});

// PTY_WRITE from main: surfaceId 기반으로 매핑
ipcRenderer.on(IPC_CHANNELS.PTY_WRITE, (_event, surfaceId: string, data: string) => {
  const ptyId = surfacePtyMap.get(surfaceId);
  if (ptyId) bridge.write(ptyId, data);
});
```

### 10.2 PanelLayoutTree 유틸리티 (P2-BUG-4)

`src/shared/panel-layout-utils.ts`:
```typescript
import type { PanelLayoutTree } from './types';

// 트리에서 panelId를 가진 leaf를 찾는다
export function findLeaf(
  tree: PanelLayoutTree,
  panelId: string,
): PanelLayoutTree | null {
  if (tree.type === 'leaf') {
    return tree.panelId === panelId ? tree : null;
  }
  return findLeaf(tree.children[0], panelId) ?? findLeaf(tree.children[1], panelId);
}

// 트리에서 panelId를 가진 leaf를 replacement로 교체한다
// 새 트리를 반환 (원본 불변)
export function replaceLeaf(
  tree: PanelLayoutTree,
  panelId: string,
  replacement: PanelLayoutTree,
): PanelLayoutTree {
  if (tree.type === 'leaf') {
    return tree.panelId === panelId ? replacement : tree;
  }
  return {
    ...tree,
    children: [
      replaceLeaf(tree.children[0], panelId, replacement),
      replaceLeaf(tree.children[1], panelId, replacement),
    ],
  };
}

// panelId를 포함하는 부모 split 노드를 찾는다
export function findParentSplit(
  tree: PanelLayoutTree,
  panelId: string,
): { direction: 'horizontal' | 'vertical'; ratio: number } | null {
  if (tree.type === 'leaf') return null;
  for (const child of tree.children) {
    if (child.type === 'leaf' && child.panelId === panelId) {
      return { direction: tree.direction, ratio: tree.ratio };
    }
    const found = findParentSplit(child, panelId);
    if (found) return found;
  }
  return null;
}

// panelId를 포함하는 split의 ratio를 변경한다
export function updateRatioForPanel(
  tree: PanelLayoutTree,
  panelId: string,
  newRatio: number,
): PanelLayoutTree {
  if (tree.type === 'leaf') return tree;
  const clamped = Math.max(0.1, Math.min(0.9, newRatio));
  // 이 split의 직접 자식에 panelId가 있는지 확인
  const isDirectChild = tree.children.some(
    (c) => c.type === 'leaf' && c.panelId === panelId,
  );
  if (isDirectChild) {
    return { ...tree, ratio: clamped };
  }
  return {
    ...tree,
    children: [
      updateRatioForPanel(tree.children[0], panelId, newRatio),
      updateRatioForPanel(tree.children[1], panelId, newRatio),
    ],
  };
}

// leaf를 제거하고 형제를 승격한다
export function removeLeaf(
  tree: PanelLayoutTree,
  panelId: string,
): PanelLayoutTree | null {
  if (tree.type === 'leaf') {
    return tree.panelId === panelId ? null : tree;
  }
  const [left, right] = tree.children;
  if (left.type === 'leaf' && left.panelId === panelId) return right;
  if (right.type === 'leaf' && right.panelId === panelId) return left;
  const newLeft = removeLeaf(left, panelId);
  if (newLeft !== left) return { ...tree, children: [newLeft ?? right, right] };
  const newRight = removeLeaf(right, panelId);
  if (newRight !== right) return { ...tree, children: [left, newRight ?? left] };
  return tree;
}
```

```typescript
case 'panel.split': {
  const { panelId, direction, newPanelType } = action.payload;
  // 패널이 속한 워크스페이스 찾기
  const ws = draft.workspaces.find((w) => {

  const newSurfaceId = crypto.randomUUID();

  // 새 패널 + 서피스 생성
  draft.panels.push({

    surfaceIds: [newSurfaceId], activeSurfaceId: newSurfaceId, isZoomed: false,
  });
  draft.surfaces.push({
    id: newSurfaceId, panelId: newPanelId,
    surfaceType: newPanelType, title: newPanelType === 'terminal' ? 'Terminal' : 'New Tab',
  });

  // 트리에서 leaf를 split으로 교체
  ws.panelLayout = replaceLeaf(ws.panelLayout, panelId, {
    type: 'split', direction, ratio: 0.5,
    children: [
      { type: 'leaf', panelId },
      { type: 'leaf', panelId: newPanelId },
    ],
  });
  break;
}

case 'panel.resize': {
  const { panelId, ratio } = action.payload;
  const ws = draft.workspaces.find((w) => findLeaf(w.panelLayout, panelId) !== null);
  if (!ws) break;
  ws.panelLayout = updateRatioForPanel(ws.panelLayout, panelId, ratio);
  break;
}

case 'panel.zoom': {
  const panel = draft.panels.find((p) => p.id === action.payload.panelId);
  if (panel) panel.isZoomed = !panel.isZoomed;
  break;
}

// P2-BUG-8 FIX: panel.close도 수정 — panelLayout 트리에서 leaf 제거
case 'panel.close': {
  const panelId = action.payload.panelId;
  const idx = draft.panels.findIndex((p) => p.id === panelId);
  if (idx === -1) break;
  // 서피스 정리
  draft.surfaces = draft.surfaces.filter((s) => s.panelId !== panelId);
  // panels 배열에서 제거
  draft.panels.splice(idx, 1);
  // panelLayout 트리에서 leaf 제거 + 형제 승격
  const ws = draft.workspaces.find((w) => findLeaf(w.panelLayout, panelId) !== null);
  if (ws) {
    const newLayout = removeLeaf(ws.panelLayout, panelId);
    if (newLayout) ws.panelLayout = newLayout;
  }
  break;
}

case 'surface.reorder': {
  const panel = draft.panels.find((p) => p.id === action.payload.panelId);
  if (!panel) break;
  const oldIndex = panel.surfaceIds.indexOf(action.payload.surfaceId);
  if (oldIndex === -1) break;
  panel.surfaceIds.splice(oldIndex, 1);
  panel.surfaceIds.splice(action.payload.newIndex, 0, action.payload.surfaceId);
  break;
}

case 'workspace.reorder': {
  const win = draft.windows.find((w) => w.id === action.payload.windowId);
  if (!win) break;
  const oldIdx = win.workspaceIds.indexOf(action.payload.workspaceId);
  if (oldIdx === -1) break;
  win.workspaceIds.splice(oldIdx, 1);
  win.workspaceIds.splice(action.payload.newIndex, 0, action.payload.workspaceId);
  break;
}
```

```typescript
function PanelLayout({ layout, ... }: PanelLayoutProps) {
  if (layout.type === 'leaf') {
    const panel = panels.find(p => p.id === layout.panelId);
    if (!panel) return null;

    // P2-BUG-5: 줌된 패널이 있으면 줌된 것만 표시

  }

  // split: CSS Grid 3칸 (P2-BUG-10: divider를 중간 셀에 배치)
  const isHorizontal = layout.direction === 'horizontal';
  const gridTemplate = isHorizontal
    ? { gridTemplateColumns: `${layout.ratio}fr 4px ${1 - layout.ratio}fr` }
    : { gridTemplateRows: `${layout.ratio}fr 4px ${1 - layout.ratio}fr` };

  return (
    <div style={{ display: 'grid', ...gridTemplate, width: '100%', height: '100%' }}>
      <PanelLayout layout={layout.children[0]} ... />
      <PanelDivider
        direction={isHorizontal ? 'vertical' : 'horizontal'}
        onDrag={(newRatio) => onResize(/* first child panelId */, newRatio)}
      />
      <PanelLayout layout={layout.children[1]} ... />
    </div>
  );
}
```

```typescript
// 키 문자열 파싱 유틸리티
export function parseKeyCombo(key: string): { ctrl: boolean; shift: boolean; alt: boolean; key: string } {
  const parts = key.split('+');
  return {
    ctrl: parts.includes('Ctrl'),
    shift: parts.includes('Shift'),
    alt: parts.includes('Alt'),
    key: parts[parts.length - 1],
  };
}

// Electron input 객체와 단축키 매칭
export function matchInput(
  input: { control: boolean; shift: boolean; alt: boolean; key: string },
  shortcuts: ShortcutDef[],
): string | null {
  for (const sc of shortcuts) {
    const combo = parseKeyCombo(sc.defaultKey);
    if (
      input.control === combo.ctrl &&
      input.shift === combo.shift &&
      input.alt === combo.alt &&
      input.key.toLowerCase() === combo.key.toLowerCase()
    ) {
      return sc.id;
    }
  }
  return null;
}
```

### shortcut-interceptor.ts (Main Process)

```typescript
// main/shortcuts/shortcut-interceptor.ts
import type { BrowserWindow } from 'electron';
import { DEFAULT_SHORTCUTS, matchInput } from '../../shared/shortcuts';

export function attachShortcutInterceptor(win: BrowserWindow): void {
  win.webContents.on('before-input-event', (event, input) => {
    // input-event 타입이 keyDown일 때만
    if (input.type !== 'keyDown') return;

    const shortcutId = matchInput(input, DEFAULT_SHORTCUTS);
    if (shortcutId) {
      event.preventDefault();
      win.webContents.send('cmux:shortcut', shortcutId);
    }
  });
}
```

### useShortcuts.ts (Renderer)

```typescript
// renderer/hooks/useShortcuts.ts
import { useEffect } from 'react';
import type { AppState } from '../../shared/types';

declare global {
  interface Window {
    cmuxIpc: {
      // ... 기존
      onShortcut: (callback: (id: string) => void) => () => void;
    };
  }
}

export function useShortcuts(
  appState: AppState | null,
  dispatch: (action: unknown) => Promise<{ ok: boolean }>,
  windowId: string | null,
  callbacks: {
    toggleSidebar: () => void;
  },
) {
  useEffect(() => {
    if (!appState || !windowId) return;

    const unsub = window.cmuxIpc.onShortcut((shortcutId: string) => {
      const activePanelId = appState.focus.activePanelId;
      const activeWsId = appState.focus.activeWorkspaceId;

      switch (shortcutId) {
        case 'newWorkspace':
          dispatch({ type: 'workspace.create', payload: { windowId } });
          break;
        case 'closeWorkspace':
          if (activeWsId) dispatch({ type: 'workspace.close', payload: { workspaceId: activeWsId } });
          break;
        case 'splitRight':
          if (activePanelId) dispatch({ type: 'panel.split', payload: { panelId: activePanelId, direction: 'horizontal', newPanelType: 'terminal' } });
          break;
        case 'splitDown':
          if (activePanelId) dispatch({ type: 'panel.split', payload: { panelId: activePanelId, direction: 'vertical', newPanelType: 'terminal' } });
          break;
        case 'toggleZoom':
          if (activePanelId) dispatch({ type: 'panel.zoom', payload: { panelId: activePanelId } });
          break;
        case 'toggleSidebar':
          callbacks.toggleSidebar();
          break;
        case 'newSurface':
          if (activePanelId) dispatch({ type: 'surface.create', payload: { panelId: activePanelId, surfaceType: 'terminal' } });
          break;
        case 'closeSurface': {
          const panel = appState.panels.find(p => p.id === activePanelId);
          if (panel) dispatch({ type: 'surface.close', payload: { surfaceId: panel.activeSurfaceId } });
          break;
        }
        // nextWorkspace, prevWorkspace, focusLeft/Right/Up/Down 등
        // 인덱스 기반 순환 로직
      }
    });

    return unsub;
  }, [appState, dispatch, windowId, callbacks]);
}
```
